19  package org.apache.hadoop.hbase.util;
20  
21  import org.apache.hadoop.hbase.classification.InterfaceAudience;
22  
23  /**
24   * Manages a singleton instance of the environment edge. This class shall
25   * implement static versions of the interface {@link EnvironmentEdge}, then
26   * defer to the delegate on invocation.
27   */
28  @InterfaceAudience.Private
29  public class EnvironmentEdgeManager {
30    private static volatile EnvironmentEdge delegate = new DefaultEnvironmentEdge();
31  
32    private EnvironmentEdgeManager() {
33  
34    }
35  
36    /**
37     * Retrieves the singleton instance of the {@link EnvironmentEdge} that is
38     * being managed.
39     *
40     * @return the edge.
41     */
42    public static EnvironmentEdge getDelegate() {
43      return delegate;
44    }
45  
46    /**
47     * Resets the managed instance to the default instance: {@link
48     * DefaultEnvironmentEdge}.
49     */
50    public static void reset() {
51      injectEdge(new DefaultEnvironmentEdge());
52    }
53  
54    /**
55     * Injects the given edge such that it becomes the managed entity. If null is
56     * passed to this method, the default type is assigned to the delegate.
57     *
58     * @param edge the new edge.
59     */
60    public static void injectEdge(EnvironmentEdge edge) {
61      if (edge == null) {
62        reset();
63      } else {
64        delegate = edge;
65      }
66    }
67  
68    /**
69     * Defers to the delegate and calls the
70     * {@link EnvironmentEdge#currentTimeMillis()} method.
71     *
72     * @return current time in millis according to the delegate.
73     */
74    public static long currentTimeMillis() {
75      return getDelegate().currentTimeMillis();
76    }
77  }
